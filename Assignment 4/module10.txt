#include <iostream>
#include <string>
using namespace std;
class Product
{
private:
    int id;
    string name;
    float price;
public:
    Product(int i, string n, float p)
    {
        id = i;
        name = n;
        price = p;
        cout << "Constructor called." << endl;
    }
    ~Product()
    {
        cout << "Destructor called." << endl;
    }
    void printDetails()
    {
        cout << "Product ID: " << id << endl;
        cout << "Product Name: " << name << endl;
        cout << "Product Price: $" << price << endl;
    }
};
int main()
{
    Product item(200, "Headphones", 79.99);
    item.printDetails();
    return 0;
}

OUTPUT:
Constructor called.
Product ID: 200
Product Name: Headphones
Product Price: $79.99
Destructor called.

knowledge check:
1.The constructor saves time because it automatically initializes the product information instead of writing the same code every time.
2.They are all part of the Product class.
